<script>

// Khởi tạo dữ liệu giả lập cho đủ 18 món ăn kèm ảnh minh họa sắc nét công cộng

const foods = [];

for (let i = 1; i <= 18; i++) {

    let categories = ['Bún cá', 'Bánh mì', 'Pizza', 'Trà sữa', 'Gà rán', 'Phở bò'];

    let currentCategory = categories[i % categories.length];

    foods.push({

        id: i,

        name: `${currentCategory} Đặc Biệt Số ${i}`,

        price: (30000 + (i * 5000)).toLocaleString('vi-VN') + ' đ',

        image: `https://picsum.photos/id/${i + 10}/400/300` // Ảnh ngẫu nhiên độ phân giải cao

    });

}

 

let filteredFoods = [...foods];

let currentPage = 1;

const itemsPerPage = 6;

 

// Hàm hiển thị danh sách món ăn

function renderFoods() {

    const grid = document.getElementById('foodGrid');

    grid.innerHTML = '';

   

    // Tính toán phân trang

    const startIndex = (currentPage - 1) * itemsPerPage;

    const endIndex = startIndex + itemsPerPage;

    const pageItems = filteredFoods.slice(startIndex, endIndex);

 

    if(pageItems.length === 0) {

        grid.innerHTML = '<p style="text-align:center; grid-column: 1/-1;">Không tìm thấy món ăn phù hợp.</p>';

        return;

    }

 

    pageItems.forEach(food => {

        grid.innerHTML += `

            <div class="food-card">

                <img src="${food.image}" class="food-img" alt="${food.name}">

                <div class="food-info">

                    <div>

                        <div class="food-title">${food.name}</div>

                        <div class="food-price">${food.price}</div>

                    </div>

                    <button class="order-btn" onclick="openModal('${food.name}', '${food.price}')">Đặt mua ngay</button>

                </div>

            </div>

        `;

    });

    renderPagination();

}

 

// Hàm hiển thị nút phân trang

function renderPagination() {

    const paginationContainer = document.getElementById('pagination');

    paginationContainer.innerHTML = '';

    const totalPages = Math.ceil(filteredFoods.length / itemsPerPage);

 

    if (totalPages <= 1) return;

 

    // Nút Previous

    paginationContainer.innerHTML += `<button ${currentPage === 1 ? 'disabled' : ''} onclick="changePage(${currentPage - 1})">Trước</button>`;

 

    // Các nút số trang

    for (let i = 1; i <= totalPages; i++) {

        paginationContainer.innerHTML += `

            <button class="${currentPage === i ? 'active' : ''}" onclick="changePage(${i})">${i}</button>

        `;

    }

 

    // Nút Next

    paginationContainer.innerHTML += `<button ${currentPage === totalPages ? 'disabled' : ''} onclick="changePage(${currentPage + 1})">Sau</button>`;

}

 

function changePage(page) {

    currentPage = page;

    renderFoods();

    window.scrollTo({top: 0, behavior: 'smooth'});

}

 

// Hàm Tìm kiếm

function searchFood() {

    const query = document.getElementById('searchInput').value.toLowerCase().trim();

    filteredFoods = foods.filter(food => food.name.toLowerCase().includes(query));

    currentPage = 1; // Reset về trang 1 khi tìm kiếm

    renderFoods();

}

 

// Hàm Đóng/Mở Modal Đặt Hàng

function openModal(name, price) {

    document.getElementById('modalTitle').innerText = "Đặt món: " + name;

    document.getElementById('modalPrice').innerText = "Giá: " + price;

   

    // Tạo nội dung chuyển khoản tự động dựa trên tên món ăn để bạn dễ quản lý tiền về

    const randomCode = Math.floor(1000 + Math.random() * 9000);

    document.getElementById('paymentNote').innerText = `DH ${randomCode} - ${name}`;

   

    document.getElementById('orderModal').style.display = 'flex';

}

 

function closeModal() {

    document.getElementById('orderModal').style.display = 'none';

}

 

// Khởi chạy ứng dụng lần đầu

renderFoods();
